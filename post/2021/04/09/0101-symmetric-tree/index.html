0101-symmetric tree

Xiu Di / 2021-04-09

题目描述

给定一个二叉树，检查它是否是镜像对称的。

示例1:

输入:

    1
   / \
  2   2
 / \ / \
3  4 4  3

输出: True

示例2:

输入:

    1
   / \
  2   2
   \   \
    3   3

代码:

# 递归
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isSame(self, lptr:TreeNode, rptr:TreeNode) -> bool:
        if not lptr and not rptr: return True
        if not lptr or not rptr or lptr.val != rptr.val: return False
        return self.isSame(lptr.left, rptr.right) and self.isSame(lptr.right, rptr.left)
    def isSymmetric(self, root: TreeNode) -> bool:
        if not root: return True
        else: return self.isSame(root.left, root.right)

# 迭代，效率较高
class Solution:
    def isSymmetric(self, root: TreeNode) -> bool:
        L = [root.left, root.right]
        while L:
            lptr, rptr = L.pop(), L.pop()
            if not lptr and not rptr: continue
            if not lptr or not rptr or lptr.val != rptr.val: return False
            L.extend([lptr.left, rptr.right, lptr.right, rptr.left])
        return True
