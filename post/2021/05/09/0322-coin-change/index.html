class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        dp = [float('inf')] * (amount + 1)
        dp[0] = 0

        coins.sort()
        for val in range(1, amount + 1):
            for coin in coins:
                if coin > val: break
                dp[val] = min(dp[val], dp[val - coin] + 1)

        return -1 if dp[amount] == float('inf') else dp[amount]
