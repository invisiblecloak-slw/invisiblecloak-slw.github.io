class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:
        hashTable = collections.defaultdict(int)

        hashTable[0] = 1

        prefixSum, cnt = 0, 0
        for num in nums:
            prefixSum += num
            if hashTable[prefixSum - k]:
                cnt += hashTable[prefixSum - k]
            hashTable[prefixSum] += 1
        return cnt
