# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def move(self) -> bool:
        if not self.rabbit.next or not self.rabbit.next.next:
            return False

        self.rabbit = self.rabbit.next.next
        self.tortoise = self.tortoise.next
        return True

    def detectCycle(self, head: ListNode) -> ListNode:
        if not head:
            return None
        
        self.tortoise = self.rabbit = head
        
        # Floydåˆ¤ç¯æ³•
        if not self.move():
            return None
        while self.rabbit != self.tortoise:
            if not self.move():
                return None

        # å¯»æ‰¾ç¯å£èŠ‚ç‚¹
        self.tortoise = head
        while self.tortoise != self.rabbit:
            self.tortoise = self.tortoise.next
            self.rabbit = self.rabbit.next

        return self.tortoise
