class Solution:
    def findUnsortedSubarray(self, nums: List[int]) -> int:
        sz = len(nums)
        if sz == 1: return 0
        # ä»å³å‘å·¦éå†æ•°ç»„ï¼Œè‹¥å­˜åœ¨åç»­å…ƒç´ å€¼å°äºå½“å‰å…ƒç´ å€¼ï¼Œé‚£ä¹ˆå½“å‰å…ƒç´ åº”çº³å…¥æ’åºèŒƒç•´
        minElement = nums[-1]
        left = -1
        for idx in range(sz-1, -1, -1):
            if minElement < nums[idx]:
                left = idx
            else:
                minElement = nums[idx]
        # ä»å·¦å‘å³éå†æ•°ç»„ï¼Œè‹¥å­˜åœ¨ä¹‹å‰å…ƒç´ å€¼å¤§äºå½“å‰å…ƒç´ å€¼ï¼Œé‚£ä¹ˆå½“å‰å…ƒç´ åº”çº³å…¥æ’åºèŒƒç•´
        right = -1
        maxElement = nums[0]
        for idx in range(sz):
            if maxElement > nums[idx]:
                right = idx
            else:
                maxElement = nums[idx]

        return 0 if left == -1 else right - left + 1
