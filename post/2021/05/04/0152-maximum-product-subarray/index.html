class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        res = maxval = minval = nums[0]
        for num in nums[1:]:
            maxval, minval = max(num, maxval * num, minval * num), min(num, minval * num, maxval * num)
            res = max(res, maxval)
        return res
