class Solution:
    def numSquares(self, n: int) -> int:
        squares = [i ** 2 for i in range(1, int(math.sqrt(n)) + 1)]

        dp = [float('inf')] * (n + 1)
        dp[0] = 0

        for i in range(1, n+1):
            for square in squares:
                if square > n:
                    break
                dp[i] = min(dp[i], 1 + dp[i - square])

class Solution:
    def numSquares(self, n: int) -> int:
        self.squares = [i ** 2 for i in range(1, int(math.sqrt(n)) + 1)]
        for k in range(1, n + 1):
            if self.canBeComposed(n, k):
                return k

    def canBeComposed(self, n: int, k: int):
        if k == 1:
            return n in self.squares

        for square in self.squares:
            if square > n: 
                break
            if self.canBeComposed(n - square, k - 1):
                return True

        return False

class Solution:
    def numSquares(self, n: int) -> int:
        squares = [i ** 2 for i in range(1, int(math.sqrt(n)) + 1)]
        
        if n in squares: return 1
        
        for square in squares:
            if square > n: break
            if n - square in squares: return 2

        while n & 3 == 0:
            n >> = 2
        if n & 7 == 7:
            return 3

        return 4
